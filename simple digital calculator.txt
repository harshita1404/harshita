// Example Main Menu Structure
#include <iostream>

void mainMenu() {
    Library lib;
    lib.loadDataFromFile(); // Load data on startup

    int choice;
    do {
        std::cout << "\n--- LIBRARY MANAGEMENT SYSTEM ---\n";
        std::cout << "1. Add New Book\n";
        std::cout << "2. Search for Book\n";
        std::cout << "3. Issue Book\n";
        std::cout << "4. Display All Books\n";
        std::cout << "5. Save & Exit\n";
        std::cout << "Enter choice: ";
        std::cin >> choice;

        switch (choice) {
            case 1: 
                // Call lib.addBook() logic
                break;
            // ... other cases
            case 5:
                lib.saveDataToFile(); // Save data before exiting
                std::cout << "Thank you! Data saved.\n";
                break;
            default:
                std::cout << "Invalid choice. Try again.\n";
        }
    } while (choice != 5);
}
